export async function getAccountId(supabase: SupabaseClient, userId: string): Promise<string> {
  const { data, error } = await supabase
    .from('investment_accounts')
    .select('id')
    .eq('user_id', userId)
    .single();
  if (error || !data) throw new Error('Could not resolve investment account');
  return data.id;
}

      instruments ( id, isin, symbol, exchange, display_name, currency, price_source_symbol )`)
    .eq('account_id', accountId);
  if (error) throw new Error(`Failed to load transactions: ${error.message}`);

  const derived = deriveHoldings(/* map rows to Transaction[] shape */);
  // Join derived {instrumentId, quantity, avgCost} back to each instrument's
  // display fields (symbol/name/exchange/currency), pulled from the same
  // query result — do NOT issue a second query per instrument (N+1 pitfall,

  return /* Holding[] */;
}

export async function getWatchlist(supabase: SupabaseClient, accountId: string): Promise<WatchlistItem[]> {
  const { data, error } = await supabase
    .from('watchlist_items')
    .select(`id, instrument_id, added_at,
      instruments ( id, isin, symbol, exchange, display_name, currency )`)
    .eq('account_id', accountId)
    .order('added_at', { ascending: false });
  if (error) throw new Error(`Failed to load watchlist: ${error.message}`);
  return /* WatchlistItem[] */;
}

export async function searchInstruments(supabase: SupabaseClient, query: string): Promise<Instrument[]> {
  if (!query.trim()) return [];
  const { data, error } = await supabase
    .from('instruments')
    .select('id, isin, symbol, exchange, display_name, asset_type, currency, price_source_symbol')
    .or(`symbol.ilike.%${query}%,display_name.ilike.%${query}%`)
    .limit(20);
  if (error) throw new Error(`Instrument search failed: ${error.message}`);
  return /* mapped Instrument[] */;
}

`'use server'` file. Every action: gets the server Supabase client (`@/utils/supabase/server`), calls `supabase.auth.getUser()` and throws `Unauthorized` if null (defense-in-depth alongside RLS, matching the pattern already used in `src/app/api/settings/keys/route.ts`), resolves `accountId` via `getAccountId`, performs the mutation against `public.transactions` or `public.watchlist_items` through the SAME cookie-bound client (never the admin client — see `src/utils/supabase/admin.ts`'s own warning comment about the SSR/service-role trap), then calls `revalidatePath('/')` and `revalidatePath('/holdings')` (add `revalidatePath('/news')` too, since it also renders the watchlist) before returning `{ success: true, ... }`.

Implement, with these Phase-2 MVP semantics (document each as an intentional simplification in a comment — full multi-lot editing is out of scope for this phase):

- `addHolding({ instrumentId, quantity, price, date, notes? })` → insert one `BUY` transaction.
- `sellHolding({ instrumentId, quantity, price, date, notes? })` → insert one `SELL` transaction. This is the PARTIAL-SELL path the roadmap calls out — do not merge this with `deleteHolding`.
- `editHolding({ instrumentId, quantity, price, date })` → MVP simplification: delete all existing transactions for `(accountId, instrumentId)` and insert a single corrected `BUY` row with the given values (resets the position rather than editing a specific historical lot). Use a Postgres transaction if the client supports it easily; otherwise sequential delete-then-insert is acceptable for Phase 2 given RLS scopes both to the same account.
- `deleteHolding({ instrumentId })` → delete ALL transactions for `(accountId, instrumentId)` (closes the position entirely; distinct from a recorded SELL — this is a correction/removal, not a market exit, and does not go through `deriveHoldings` history).
- `recordSplit({ instrumentId, additionalQuantity, date, notes? })` → insert one `SPLIT` transaction, `price: null`.
- `recordBonus({ instrumentId, additionalQuantity, date, notes? })` → insert one `BONUS` transaction, `price: null`.
- `addToWatchlist({ instrumentId })` → insert into `watchlist_items`; catch the unique-constraint violation (already-watching) and return `{ success: false, error: 'Already on watchlist' }` instead of throwing.
- `removeFromWatchlist({ watchlistItemId })` → delete by id (RLS scopes to the caller's account already, no extra WHERE needed beyond the id).

Validate numeric inputs (`quantity > 0`, `price >= 0` for BUY/SELL) in the action before insert — this mirrors the DB CHECK constraints from plan 02-01 but gives a friendlier error message than a raw Postgres constraint violation.
