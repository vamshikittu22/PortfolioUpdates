export interface RefreshSummary {
  instrumentsConsidered: number;
  instrumentsFetched: number;
  instrumentsSkippedDedup: number;
  succeeded: number;
  failed: number;
  fxUpdated: boolean;
  timestamp: string;
}

export async function refreshAllPrices(admin: SupabaseClient): Promise<RefreshSummary>
```

Steps (use the admin client for every query in this function — this is the documented cross-user "scheduled/admin operation" case from AUTH-05; a cookie-bound client would only see the calling user's own transactions/watchlist rows, undercounting the shared cache's needs):

1. Discover instrument ids to price: `SELECT DISTINCT instrument_id FROM transactions` UNION `SELECT DISTINCT instrument_id FROM watchlist_items` (two queries + `Set` dedup in JS is fine — no need for raw SQL). Join to `instruments` for `price_source_symbol` and `currency`.
2. Fetch existing `price_cache` rows for those instrument ids (`updated_at`). For each instrument, apply `shouldSkipRefresh(existingRow?.updated_at ? new Date(existingRow.updated_at) : null, new Date())` from `@/lib/prices/ingest` — build the subset that actually needs fetching; count the rest as `instrumentsSkippedDedup`.
3. Call `fetchPrices(symbolsToFetch)` from `@/lib/prices/fetch-prices` for the subset (map `price_source_symbol → instrument_id` to translate results back).
4. For each fetched instrument, partition into success/failure:
   - **Success**: build a full row `{ instrument_id, symbol, price, change_pct: changePct, source: 'yahoo-finance', fetch_error: null, corporate_action_flag: detectCorporateAction(changePct), updated_at: new Date().toISOString() }` and collect for a batch `.upsert(rows, { onConflict: 'instrument_id' })`.
   - **Failure**: do NOT include this instrument in the upsert batch. Instead, `UPDATE price_cache SET fetch_error = $1 WHERE instrument_id = $2` (only touches `fetch_error`, leaves `price`/`change_pct`/`updated_at`/`corporate_action_flag` exactly as they were — this is the "never clobber a good price with a failure" guarantee). If that `UPDATE` affects zero rows (brand-new instrument, never priced before), fall back to `INSERT { instrument_id, symbol, price: null, change_pct: null, source: null, fetch_error: <error>, corporate_action_flag: false, updated_at: now }` — the ONLY case where a NULL price row is written is "this instrument has never once been successfully priced," which is honest, not fabricated.
5. FX: call `fetchFXRate('USD', 'INR')` from `@/lib/prices/fx-rates`. On success, `upsert({ pair: 'USD_INR', rate, fetch_error: null, updated_at: now })` into `fx_cache`. On failure, same partial-update-or-insert pattern as price_cache (update only `fetch_error`, preserve the last-known-good `rate`; insert a `rate: null` row only if none exists yet).
6. Return the `RefreshSummary` counts.

Document in a comment why FX is refreshed unconditionally on every call (not dedup'd like prices) — it's a single cheap extra request per research's Open Question 4 recommendation, and keeping it simple avoids a second dedup timer.

// src/app/api/prices/refresh/route.ts
import { NextResponse } from 'next/server';
import { isAuthorizedRefreshRequest } from '@/lib/prices/ingest';
import { refreshAllPrices } from '@/lib/prices/refresh-service';
import { createAdminClient } from '@/utils/supabase/admin';

export async function POST(request: Request) {
  const authHeader = request.headers.get('authorization');
  if (!isAuthorizedRefreshRequest(authHeader, process.env.PRICE_REFRESH_SECRET ?? '')) {
    return NextResponse.json({ error: 'Unauthorized' }, { status: 401 });
  }

  try {
    const summary = await refreshAllPrices(createAdminClient());
    return NextResponse.json({ success: true, ...summary });
  } catch (error) {
    console.error('Price refresh failed:', error);
    return NextResponse.json(
      { error: error instanceof Error ? error.message : 'Unknown error' },
      { status: 500 }
    );
  }
}
```

Note the guard check happens BEFORE `createAdminClient()`/`refreshAllPrices()` are ever called — an unauthorized request never touches Supabase, matching the pure `isAuthorizedRefreshRequest` predicate tested in 03-02.

This does NOT need a `user_setup` entry — it's self-generated, not sourced from a provider.

'use server';

import { createClient } from '@/utils/supabase/server';
import { createAdminClient } from '@/utils/supabase/admin';
import { refreshAllPrices, type RefreshSummary } from '@/lib/prices/refresh-service';
import { revalidatePath } from 'next/cache';

export async function refreshPricesNow(): Promise<
  { success: true; summary: RefreshSummary } | { success: false; error: string }
> {
  const supabase = await createClient();
  const { data: { user } } = await supabase.auth.getUser();
  if (!user) return { success: false, error: 'Unauthorized' };

  try {
    // Elevated write access is required here (price_cache/fx_cache have no
    // authenticated write policy per Phase 1's rls_fixes migration) — this mirrors
    // the cron path exactly, just triggered by a signed-in user instead of pg_cron.
    // The admin client is used ONLY for the refresh write, never returned to the
    // caller or exposed beyond this server-only function.
    const summary = await refreshAllPrices(createAdminClient());
    revalidatePath('/');
    revalidatePath('/holdings');
    return { success: true, summary };
  } catch (error) {
    return { success: false, error: error instanceof Error ? error.message : 'Refresh failed' };
  }
}
